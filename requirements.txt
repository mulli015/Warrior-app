BASE_DIR = os.path.abspath(os.path.dirname(__file__))
DB_PATH = os.path.join(BASE_DIR, 'lightning_warriors.db')

app = Flask(__name__)

db = SQLAlchemy(app)

team_members = db.Table(
    'team_members',
    db.Column('user_id', db.Integer, db.ForeignKey('user.id'), primary_key=True),
    db.Column('team_id', db.Integer, db.ForeignKey('team.id'), primary_key=True),
)


class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    full_name = db.Column(db.String(120), nullable=False)
    email = db.Column(db.String(120), unique=True, nullable=False)
    password_hash = db.Column(db.String(255), nullable=False)
    phone = db.Column(db.String(30))
    branch = db.Column(db.String(50))
    role = db.Column(db.String(20), default='member')  # admin, manager, member
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

    event_registrations = db.relationship('EventRegistration', backref='user', lazy=True)
    attendances = db.relationship('AttendanceRecord', backref='user', lazy=True, foreign_keys='AttendanceRecord.user_id')
    teams = db.relationship('Team', secondary=team_members, back_populates='members')

    def set_password(self, password: str) -> None:
        self.password_hash = generate_password_hash(password)

    def check_password(self, password: str) -> bool:
        return check_password_hash(self.password_hash, password)


class Team(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(120), nullable=False, unique=True)
    season = db.Column(db.String(40))
    division = db.Column(db.String(40))
    description = db.Column(db.Text)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

    members = db.relationship('User', secondary=team_members, back_populates='teams')
    events = db.relationship('Event', backref='team', lazy=True, cascade='all, delete-orphan')
    messages = db.relationship('Message', backref='team', lazy=True, cascade='all, delete-orphan')


class Event(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(150), nullable=False)
    event_type = db.Column(db.String(50), nullable=False)
    start_time = db.Column(db.DateTime, nullable=False)
    end_time = db.Column(db.DateTime, nullable=False)
    location = db.Column(db.String(180))
    notes = db.Column(db.Text)
    team_id = db.Column(db.Integer, db.ForeignKey('team.id'), nullable=False)
    created_by_user_id = db.Column(db.Integer, db.ForeignKey('user.id'))
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

    registrations = db.relationship('EventRegistration', backref='event', lazy=True, cascade='all, delete-orphan')
    messages = db.relationship('Message', backref='event', lazy=True, cascade='all, delete-orphan')
    attendance_records = db.relationship('AttendanceRecord', backref='event', lazy=True, cascade='all, delete-orphan')


class EventRegistration(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)
    event_id = db.Column(db.Integer, db.ForeignKey('event.id'), nullable=False)
    status = db.Column(db.String(20), nullable=False, default='registered')  # registered, maybe, not_attending
    registered_at = db.Column(db.DateTime, default=datetime.utcnow)

    __table_args__ = (db.UniqueConstraint('user_id', 'event_id', name='uq_user_event'),)


class AttendanceRecord(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    event_id = db.Column(db.Integer, db.ForeignKey('event.id'), nullable=False)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)
    status = db.Column(db.String(20), nullable=False, default='unmarked')  # present, absent, late, excused, unmarked
    minutes_late = db.Column(db.Integer)
    notes = db.Column(db.String(255))
    marked_by_user_id = db.Column(db.Integer, db.ForeignKey('user.id'))
    updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    __table_args__ = (db.UniqueConstraint('event_id', 'user_id', name='uq_attendance_event_user'),)


class Message(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    content = db.Column(db.Text, nullable=False)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)
    team_id = db.Column(db.Integer, db.ForeignKey('team.id'), nullable=True)
    event_id = db.Column(db.Integer, db.ForeignKey('event.id'), nullable=True)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

    author = db.relationship('User', backref='messages', foreign_keys=[user_id])


ATTENDANCE_LABELS = {
    'present': 'Present',
    'late': 'Late',
    'absent': 'Absent',
    'excused': 'Excused',
    'unmarked': 'Unmarked',
}

RSVP_LABELS = {
    'registered': 'Going',
    'maybe': 'Maybe',
    'not_attending': 'Not attending',
}


def current_user():
    user_id = session.get('user_id')
    if not user_id:
        return None
    return db.session.get(User, user_id)


@app.context_processor
def inject_user():
    return {
        'current_user': current_user(),
        'attendance_labels': ATTENDANCE_LABELS,
        'rsvp_labels': RSVP_LABELS,
    }


def login_required(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        if not current_user():
            flash('Please sign in first.', 'warning')
            return redirect(url_for('login'))
        return func(*args, **kwargs)
    return wrapper


def role_required(*roles):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            user = current_user()
            if not user:
                flash('Please sign in first.', 'warning')
                return redirect(url_for('login'))
            if user.role not in roles:
                abort(403)
            return func(*args, **kwargs)
        return wrapper
    return decorator


def team_access_required(team):
    user = current_user()
    if not user:
        abort(403)
    if user.role in ['admin', 'manager']:
        return True
    if user in team.members:
        return True
    abort(403)


def get_or_create_attendance(event_id: int, user_id: int) -> AttendanceRecord:
    record = AttendanceRecord.query.filter_by(event_id=event_id, user_id=user_id).first()
    if not record:
        record = AttendanceRecord(event_id=event_id, user_id=user_id, status='unmarked')
        db.session.add(record)
        db.session.flush()
    return record


def event_roster_rows(event: Event):
    registrations = {
        reg.user_id: reg for reg in EventRegistration.query.filter_by(event_id=event.id).all()
    }
    attendance = {
        rec.user_id: rec for rec in AttendanceRecord.query.filter_by(event_id=event.id).all()
    }
    team_members_sorted = sorted(event.team.members, key=lambda m: m.full_name.lower())
    rows = []
    for member in team_members_sorted:
        rows.append({
            'member': member,
            'registration': registrations.get(member.id),
            'attendance': attendance.get(member.id),
        })
    return rows


def attendance_summary(rows):
    counts = {key: 0 for key in ATTENDANCE_LABELS.keys()}
    for row in rows:
        status = 'unmarked'
        if row['attendance']:
            status = row['attendance'].status
        counts[status] = counts.get(status, 0) + 1
    return counts

@app.route('/register', methods=['GET', 'POST'])
def register():
    if request.method == 'POST':
        full_name = request.form['full_name'].strip()
        email = request.form['email'].strip().lower()
        password = request.form['password']
        phone = request.form.get('phone', '').strip()
        branch = request.form.get('branch', '').strip()

        if User.query.filter_by(email=email).first():
            flash('An account with that email already exists.', 'danger')
            return redirect(url_for('register'))

        user = User(full_name=full_name, email=email, phone=phone, branch=branch)
        if User.query.count() == 0:
            user.role = 'admin'
        user.set_password(password)
        db.session.add(user)
        db.session.commit()
        session['user_id'] = user.id
        flash('Registration complete. Welcome to Lightning Warriors Hockey.', 'success')
        return redirect(url_for('dashboard'))
    return render_template('register.html')


@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        email = request.form['email'].strip().lower()
        password = request.form['password']
        user = User.query.filter_by(email=email).first()
        if not user or not user.check_password(password):
            flash('Invalid email or password.', 'danger')
            return redirect(url_for('login'))
        session['user_id'] = user.id
        flash(f'Welcome back, {user.full_name}.', 'success')
        return redirect(url_for('dashboard'))
    return render_template('login.html')

@app.route('/dashboard')
@login_required
def dashboard():
    user = current_user()
    teams = user.teams
    query = Event.query.join(Team)
    if user.role not in ['admin', 'manager']:
        query = query.filter(Team.members.any(id=user.id))
    upcoming_events = query.order_by(Event.start_time.asc()).limit(15).all()

    my_attendance = []
    if user.role in ['admin', 'manager']:
        recent_events = Event.query.order_by(Event.start_time.desc()).limit(5).all()
    else:
        recent_events = query.order_by(Event.start_time.desc()).limit(5).all()
    for event in recent_events:
        registration = EventRegistration.query.filter_by(event_id=event.id, user_id=user.id).first()
        attendance = AttendanceRecord.query.filter_by(event_id=event.id, user_id=user.id).first()
        my_attendance.append({
            'event': event,
            'registration': registration,
            'attendance': attendance,
        })
    return render_template('dashboard.html', teams=teams, upcoming_events=upcoming_events, my_attendance=my_attendance)

@app.route('/events/<int:event_id>/attendance', methods=['POST'])
@role_required('admin', 'manager')
def update_attendance(event_id):
    event = db.session.get(Event, event_id) or abort(404)
    member_id = int(request.form['user_id'])
    status = request.form.get('attendance_status', 'unmarked')
    minutes_late_raw = request.form.get('minutes_late', '').strip()
    notes = request.form.get('notes', '').strip()

    if not any(member.id == member_id for member in event.team.members):
        abort(400)

    record = get_or_create_attendance(event.id, member_id)
    record.status = status
    record.minutes_late = int(minutes_late_raw) if minutes_late_raw else None
    record.notes = notes or None
    record.marked_by_user_id = current_user().id
    db.session.commit()
    flash('Attendance updated.', 'success')
    return redirect(url_for('event_detail', event_id=event.id))


@app.route('/events/<int:event_id>/attendance/bulk', methods=['POST'])
@role_required('admin', 'manager')
def bulk_attendance(event_id):
    event = db.session.get(Event, event_id) or abort(404)
    action = request.form.get('action')
    if action not in {'present_all_going', 'reset_all'}:
        abort(400)

    if action == 'present_all_going':
        registrations = EventRegistration.query.filter_by(event_id=event.id, status='registered').all()
        for reg in registrations:
            record = get_or_create_attendance(event.id, reg.user_id)
            record.status = 'present'
            record.minutes_late = None
            record.notes = 'Bulk marked from going list'
            record.marked_by_user_id = current_user().id
        flash('Marked all players who RSVP’d Going as present.', 'success')
    else:
        AttendanceRecord.query.filter_by(event_id=event.id).delete()
        flash('Attendance marks reset for this event.', 'info')

    db.session.commit()
    return redirect(url_for('event_detail', event_id=event.id))


@app.route('/events/<int:event_id>/attendance/export')
@role_required('admin', 'manager')
def export_attendance(event_id):
    event = db.session.get(Event, event_id) or abort(404)
    rows = event_roster_rows(event)

    buffer = io.StringIO()
    writer = csv.writer(buffer)
    writer.writerow(['Team', 'Event', 'Member', 'Email', 'Branch', 'RSVP', 'Attendance', 'Minutes Late', 'Notes'])
    for row in rows:
        registration = row['registration']
        attendance = row['attendance']
        writer.writerow([
            event.team.name,
            event.title,
            row['member'].full_name,
            row['member'].email,
            row['member'].branch or '',
            RSVP_LABELS.get(registration.status, 'No response') if registration else 'No response',
            ATTENDANCE_LABELS.get(attendance.status, 'Unmarked') if attendance else 'Unmarked',
            attendance.minutes_late if attendance and attendance.minutes_late is not None else '',
            attendance.notes if attendance and attendance.notes else '',
        ])

    response = make_response(buffer.getvalue())
    response.headers['Content-Type'] = 'text/csv'
    response.headers['Content-Disposition'] = f'attachment; filename=attendance-event-{event.id}.csv'
    return response
